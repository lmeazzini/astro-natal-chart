% Section formatting — Astro Essence style
\newcommand{\chartsection}[1]{%
  \section*{\color{primary}\LARGE\bfseries #1}%
  \addcontentsline{toc}{section}{#1}%
  \vspace{0.5cm}%
}

% Subsection formatting
\newcommand{\chartsubsection}[1]{%
  \subsection*{\color{textdark}\Large\bfseries #1}%
  \vspace{0.3cm}%
}

% Planet name with colored symbol
\newcommand{\planetname}[2]{%
  \textbf{\color{primary}#1} \textit{\color{textmuted}(#2)}%
}

% House number formatting
\newcommand{\housename}[1]{%
  \textbf{\color{secondary}Casa #1}%
}

% Aspect formatting
\newcommand{\aspectname}[3]{%
  \textbf{\color{primary}#1} \color{textmuted}{#2} \textbf{\color{primary}#3}%
}

% Dignity formatting with colored badges
\newcommand{\dignity}[2]{%
  \textcolor{#1}{\textbf{#2}}%
}

% Interpretation text box — Premium card style
\newcommand{\interpretation}[1]{%
  \begin{tcolorbox}[
    enhanced,
    colback=bgcard,        % Soft background
    colframe=border,       % Subtle border
    arc=5mm,               % ~12px border radius
    boxrule=1.5pt,
    left=16pt,             % 16px padding (md spacing)
    right=16pt,
    top=16pt,
    bottom=16pt,
    drop shadow={
      opacity=0.04,
      shadow xshift=0pt,
      shadow yshift=-2pt,
      fill=textdark
    }
  ]
  \color{textdark}
  #1
  \end{tcolorbox}
  \vspace{0.5cm}
}

% Chart info box — Premium card with Astro Essence styling
\newcommand{\chartinfo}[4]{%
  \begin{tcolorbox}[
    enhanced,
    colback=accent!30,     % Soft ivory background
    colframe=primary,
    arc=8mm,               % 20px border radius
    boxrule=1.5pt,
    title={
      \Large\bfseries Dados do Mapa
    },
    fonttitle=\bfseries,
    coltitle=white,
    colbacktitle=primary,
    left=20pt,             % Generous padding
    right=20pt,
    top=20pt,
    bottom=20pt,
    drop shadow={
      opacity=0.08,
      shadow xshift=0pt,
      shadow yshift=-4pt,
      fill=textdark
    }
  ]
  \color{textdark}
  \textbf{Nome:} \textcolor{textdark}{#1} \\[0.3cm]
  \textbf{Data e Hora:} \textcolor{textmuted}{#2} \\[0.3cm]
  \textbf{Local:} \textcolor{textmuted}{#3} \\[0.3cm]
  \textbf{Coordenadas:} \textcolor{textmuted}{#4}
  \end{tcolorbox}
}

% Key insight box (for important interpretations)
\newcommand{\keyinsight}[1]{%
  \begin{tcolorbox}[
    enhanced,
    colback=primary!10,
    colframe=primary,
    arc=5mm,
    boxrule=1.5pt,
    left=16pt,
    right=16pt,
    top=16pt,
    bottom=16pt,
    drop shadow={
      opacity=0.06,
      shadow xshift=0pt,
      shadow yshift=-3pt,
      fill=primary
    }
  ]
  \color{textdark}
  \textbf{✨ Insight Principal:} #1
  \end{tcolorbox}
  \vspace{0.3cm}
}

% Highlight box (for warnings or important notes)
\newcommand{\highlight}[1]{%

    enhanced,
    colback=warning!10,
    colframe=warning,
    arc=5mm,
    boxrule=1.5pt,
    left=16pt,
    right=16pt,
    top=12pt,
    bottom=12pt
  ]
  \color{textdark}
  \textbf{⚠️ Nota:} #1
  \end{tcolorbox}
  \vspace{0.3cm}
}
